use Illuminate\Support\Facades\File;

use App\Models\Profile;

public function storeProfile(Request $request){

    $all_data=new Profile;
                                                        
    $all_data ->name= $request->name;
                                                        
    if($request->hasFile('image')) {
        $file=$request->file('image');
        $name=$file->getClientOriginalName();
        $request->file('image')->storeAs('images/blog/',$name);
        $file->move(public_path('images/blog/'),$name);
        $all_data->image = $name;
    }
                                                        
    $all_data->save();
                                                        
    return redirect()->back()->with('message', 'Added Successfully!!');
}                                                  
                                                        
public function indexProfile(){

    $all_data=Profile::select('*')
                ->orderBy('id', 'DESC')
                ->get();
                                                          
    return view('profile.indexProfile',compact('all_data'));
}                                                                                                           
                                                        
public function showProfile($id){

    $data=Profile::find($id);
                                                                                                                    
    return view('profile.showProfile', compact('data'));
}                                                
                                                        
public function editProfile($id){

    $data=Profile::find($id);
                                                        
    return view('profile.editProfile', compact('data'));
}                                                                                                         
                                                        
public function updateProfile(Request $request, $id){

    $data=Profile::find($id);
                                                        
    $data->name= $request->name;
                                                                
    if($request->hasfile('image')){
        $destination = 'images/blog/'.$data->image;
        if(File::exists($destination)){
            File::delete($destination);
        }
        $file=$request->file('image');
        $name=$file->getClientOriginalName();
        $request->file('image')->storeAs('images/blog/',$name);
        $file->move(public_path('images/blog/'),$name);
        $data->image = $name;
    }
                                                                
    $data->update();
                                                        
    return redirect()->to('/showProfile')->with('message', 'Updated Successfully');  
}                                                                                                         
                                                        
public function deleteProfile($id){

    $data = Profile::find($id);
                        
    $destination = 'images/blog/'.$data->image;
    if(File::exists($destination)){
        File::delete($destination);
    }          
                        
    $data->delete();
                                                        
    return redirect()->back()->with('message', 'Deleted Successfully');
}
